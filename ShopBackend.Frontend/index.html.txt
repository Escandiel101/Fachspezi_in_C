<script>
    const API_URL = "http://localhost:5139/api";

    function show(id) {
        document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
        document.getElementById(id).classList.add('active');
        document.getElementById('status-msg').innerText = "Bereit für " + id;
    }

    async function fetchData(endpoint, tableId) {
        const tbody = document.querySelector(`#${tableId} tbody`);
        const status = document.getElementById('status-msg');
        status.innerText = `Lade ${endpoint}...`;
        tbody.innerHTML = "";

        try {
            // Hinweis: Falls du [Authorize] noch aktiv hast, wird das hier fehlschlagen (401)
            const response = await fetch(`${API_URL}/${endpoint}`);
            if (!response.ok) throw new Error("Server antwortet mit " + response.status);
            
            const data = await response.json();
            if (data.length === 0) {
                tbody.innerHTML = "<tr><td colspan='3'>Keine Daten vorhanden.</td></tr>";
            } else {
                data.forEach(item => {
                    const row = document.createElement('tr');
                    // Dynamisches Mapping je nach Datenstruktur
                    row.innerHTML = `
                        <td>${item.id || item.entityId || '-'}</td>
                        <td>${item.name || item.firstName || item.action || '-'}</td>
                        <td>${item.email || item.status || item.changedBy || '-'}</td>
                    `;
                    tbody.innerHTML += row.outerHTML;
                });
            }
            status.innerText = "Daten erfolgreich geladen.";
        } catch (err) {
            status.innerText = "Fehler: " + err.message + " (Check CORS & Auth!)";
            status.style.color = "red";
        }
    }
</script>
